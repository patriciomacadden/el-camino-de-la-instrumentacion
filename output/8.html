function createPresentation(title) {
  return {
    title: title,
    slides: [],
    
    addSlide(content) {
      this.slides.push(content);
      return this;
    }
  };
}

def process_markdown(content):
    """Convert markdown to HTML"""
    lines = content.split('\n')
    processed = []
    
    for line in lines:
        if line.startswith('#'):
            processed.append(f"
{line[1:].strip()}
")
        else:
            processed.append(f"
{line}
")
    
    return '\n'.join(processed)
