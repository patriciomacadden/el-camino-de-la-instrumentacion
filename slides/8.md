```python
def process_markdown(content):
    """Convert markdown to HTML"""
    lines = content.split('\n')
    processed = []
    
    for line in lines:
        if line.startswith('#'):
            processed.append(f"<h1>{line[1:].strip()}</h1>")
        else:
            processed.append(f"<p>{line}</p>")
    
    return '\n'.join(processed)
```
